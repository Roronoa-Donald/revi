Tableaux & Listes

Array & List<T>
// Tableau (taille fixe)
int[] tab = new int[5];
int[] nums = { 1, 2, 3 };
tab.Length;  // taille

// Liste (taille dynamique)
List<string> noms = new List<string>();
noms.Add("Alice");
noms.Remove("Alice");
noms.Contains("Bob");  // bool
noms.Count;               // nombre d'éléments

// LINQ
var pairs = nums.Where(n => n % 2 == 0);
var doubles = nums.Select(n => n * 2);
var tri = nums.OrderBy(n => n);

Méthodes

Syntaxe
// Void (pas de retour)
static void Saluer(string nom) {
    Console.WriteLine($"Bonjour {nom}");
}

// Avec retour
static int Additionner(int a, int b) {
    return a + b;
}

// Expression-body
static int Doubler(int n) => n * 2;

// Paramètre par défaut
static void Log(string msg, string level = "INFO") { }

// Surcharge
static int Max(int a, int b) => a > b ? a : b;
static double Max(double a, double b) => a > b ? a : b;

Dictionnaires

Dictionary<TKey, TValue>
var dict = new Dictionary<string, int>();
dict.Add("age", 25);
dict["age"] = 26;         // ajoute ou met à jour
dict.ContainsKey("age");  // true
dict.Remove("age");

// Accès sécurisé
if (dict.TryGetValue("age", out int val))
    Console.WriteLine(val);

// Itération
foreach (var kvp in dict)
    Console.WriteLine($"{kvp.Key}: {kvp.Value}");

Console I/O

Entrée / Sortie
Console.Write("Nom : ");       // sans retour à la ligne
Console.WriteLine("Bonjour");  // avec retour à la ligne
string input = Console.ReadLine();

// Conversion sécurisée
if (int.TryParse(input, out int n))
    Console.WriteLine(n);

// Formatage
Console.WriteLine($"{prix:C2}");   // monnaie
Console.WriteLine($"{pct:P1}");    // pourcentage

// Couleurs
Console.ForegroundColor = ConsoleColor.Red;

Switch & Enums

Switch & Enum
// Switch classique
switch (jour) {
    case "Lundi": Console.WriteLine("Courage"); break;
    case "Vendredi": Console.WriteLine("Bientôt!"); break;
    default: Console.WriteLine("Patience"); break;
}

// Switch expression (C# 8+)
string emoji = jour switch {
    "Lundi" => "😴",
    "Vendredi" => "🎉",
    _ => "📅"
};

// Enum
enum Saison { Printemps, Ete, Automne, Hiver }
Saison s = Saison.Ete;
int val = (int)s;  // 1

Gestion d'Erreurs

try / catch / finally / throw
try {
    int n = int.Parse("abc");
}
catch (FormatException ex) {
    Console.WriteLine(ex.Message);
}
catch (Exception ex) {
    Console.WriteLine("Erreur générale");
}
finally {
    Console.WriteLine("Toujours exécuté");
}

// Lancer une exception
throw new ArgumentException("Paramètre invalide");

// using = Dispose automatique
using var reader = new StreamReader("f.txt");

POO — Programmation Orientée Objet

Classe, Propriétés, Constructeur, Héritage
public class Animal {
    public string Nom { get; set; }
    public int Age { get; private set; }

    public Animal(string nom, int age) {
        Nom = nom; Age = age;
    }
    public void Parler() =>
        Console.WriteLine($"{Nom} fait un bruit");
}

// Héritage
public class Chien : Animal {
    public string Race { get; set; }
    public Chien(string nom, int age, string race)
        : base(nom, age) { Race = race; }
}

// Instanciation
var rex = new Chien("Rex", 3, "Berger");
rex.Parler();  // Rex fait un bruit
