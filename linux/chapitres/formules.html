Cheat Sheet Linux

Toutes les commandes essentielles sur une seule page. Imprime-la !

Navigation

pwd              # Répertoire courant
ls -la           # Liste détaillée + cachés
ls -lh           # Tailles lisibles
cd /etc           # Aller dans /etc
cd ..             # Répertoire parent
cd ~              # Répertoire home
cd -              # Répertoire précédent

Affichage

cat fichier      # Afficher le contenu
less fichier     # Paginer le contenu
head -20 f      # 20 premières lignes
tail -f f       # Suivre en temps réel
echo $USER       # Afficher une variable
man cmd          # Manuel d'une commande
history           # Historique des commandes

Fichiers & Dossiers

touch f.txt      # Créer fichier vide
mkdir -p a/b/c  # Créer dossiers récursifs
cp -r src dest   # Copier récursivement
mv old new       # Déplacer/renommer
rm -r dossier   # Supprimer récursif
ln -s src lien  # Lien symbolique
tar -czf a.tgz d # Archiver + gzip
tar -xzf a.tgz  # Extraire

Permissions

# r=4  w=2  x=1
chmod 755 f      # rwxr-xr-x
chmod 644 f      # rw-r--r--
chmod +x f      # Ajouter exécution
chmod -R 755 d  # Récursif
chown user:grp f # Changer propriétaire
chgrp grp f     # Changer groupe
umask 022        # Masque par défaut

Recherche

find / -name "*.log"  # Par nom
find . -type f -size +10M # > 10 Mo
grep -ri "err" .     # Récursif insensible
grep -n "pat" f      # Avec n° de ligne
locate nom             # Recherche rapide (BDD)
which cmd              # Chemin d'un programme
wc -l f               # Compter les lignes

Pipes & Redirections

cmd1 | cmd2      # Pipe : sortie → entrée
cmd > f          # Écraser dans fichier
cmd >> f         # Ajouter au fichier
cmd 2> err.log   # Rediriger les erreurs
cmd &> all.log   # Tout rediriger
cmd 2> /dev/null # Ignorer les erreurs
sort f | uniq -c # Trier + compter

Nano

Ctrl+O  Sauvegarder
Ctrl+X  Quitter
Ctrl+K  Couper la ligne
Ctrl+U  Coller
Ctrl+W  Rechercher
Ctrl+\  Rechercher & remplacer
Alt+U   Annuler

Vim essentiels

i       Mode insertion
Esc     Mode normal
:w      Sauvegarder
:q!     Quitter sans sauvegarder
:wq     Sauvegarder et quitter
dd      Supprimer une ligne
yy / p  Copier / Coller
/mot    Rechercher
u       Annuler

Utilisateurs & Groupes

useradd -m -s /bin/bash u
passwd u          # Mot de passe
usermod -aG sudo u # Ajouter au grp
userdel -r u     # Supprimer + home
groupadd grp      # Créer un groupe
id u              # Voir UID/GID
groups u          # Voir les groupes
su -              # Devenir root
sudo cmd         # Exécuter en root
visudo            # Éditer sudoers

Réseau

ip addr           # Voir les interfaces
ip route          # Table de routage
ping -c4 8.8.8.8 # Test connectivité
ss -tulnp        # Ports TCP/UDP ouverts
hostnamectl       # Infos hostname
# Config : /etc/network/interfaces
# DNS    : /etc/resolv.conf
# Hosts  : /etc/hosts

SSH

ssh user@host           # Se connecter
ssh -p 2222 user@host  # Port personnalisé
scp f user@host:path     # Copier via SSH
ssh-keygen -t ed25519    # Générer une clé
ssh-copy-id user@host   # Copier clé publique
# Config : /etc/ssh/sshd_config
# PermitRootLogin no
# PasswordAuthentication no

Services (systemctl)

systemctl status ssh   # État
systemctl start ssh    # Démarrer
systemctl stop ssh     # Arrêter
systemctl restart ssh  # Redémarrer
systemctl enable ssh   # Auto au boot
systemctl disable ssh  # Pas au boot

Paquets (apt)

sudo apt update            # MAJ liste
sudo apt upgrade           # MAJ paquets
sudo apt install pkg      # Installer
sudo apt remove pkg       # Désinstaller
sudo apt purge pkg        # Supprimer + config
apt search mot            # Chercher un paquet
dpkg -l                   # Lister les installés

Infos Système

uname -a          # Infos kernel
hostname           # Nom de la machine
uptime             # Temps de fonctionnement
free -h            # Mémoire RAM
df -h              # Espace disque
du -sh dossier    # Taille d'un dossier
top / htop        # Processus en cours
ps aux             # Liste des processus
