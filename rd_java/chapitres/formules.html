Variables, Types & Opérateurs

// 8 types primitifs
byte    b  = 127;       // 1 octet  (-128 à 127)
short   s  = 32000;     // 2 octets
int     i  = 42;        // 4 octets (le plus courant)
long    l  = 100000L;   // 8 octets (suffixe L)
float   f  = 3.14f;     // 4 octets (suffixe f)
double  d  = 3.14;      // 8 octets (par défaut)
boolean ok = true;     // true / false
char    c  = 'A';       // 2 octets (Unicode)

// Constante
final double PI = 3.14159;

// Opérateurs arithmétiques : + - * / %

// 5.0 / 2 = 2.5 (au moins un double)

// Casting
double x = 9.7;
int y = (int) x;  // y = 9 (troncature)

// Incrémentation
i++;  // post : utilise puis +1
++i;  // pré  : +1 puis utilise

Boucles

// for classique
for (int i = 0; i < 10; i++) {
    System.out.println(i);
}

// for-each (tableau ou collection)
for (String fruit : fruits) {
    System.out.println(fruit);
}

// while
while (compteur > 0) {
    System.out.println(compteur);
    compteur--;
}

// do-while (au moins 1 exécution)
do {
    System.out.print("Choix : ");
    choix = sc.nextInt();
} while (choix != 0);

Fonctions & Procédures

// Procédure (void — ne retourne rien)
public static void afficher(String msg) {
    System.out.println(msg);
}

// Fonction (retourne une valeur)
public static int additionner(int a, int b) {
    return a + b;
}

// Surcharge (overloading) — même nom, paramètres différents
public static int add(int a, int b) { return a + b; }
public static double add(double a, double b) { return a + b; }
public static int add(int a, int b, int c) { return a+b+c; }

// Appel
int resultat = additionner(5, 3);
afficher("Résultat : " + resultat);

Classes & Objets

// Définir une classe
public class Voiture {
    String marque;
    int vitesse;

    // Constructeur
    public Voiture(String marque, int vitesse) {
        this.marque = marque;
        this.vitesse = vitesse;
    }

    // Méthode
    void accelerer() {
        this.vitesse += 10;
    }
}

// Créer un objet
Voiture v = new Voiture("Toyota", 0);
v.accelerer();
System.out.println(v.vitesse);  // 10

Encapsulation

// Modificateurs d'accès
// private    → classe seulement
// (default)  → package seulement
// protected  → package + sous-classes
// public     → partout

public class Personne {
    private String nom;
    private int age;
    private static int compteur = 0;  // partagé

    public Personne(String nom, int age) {
        this.nom = nom;
        setAge(age);
        compteur++;
    }

    // Getter
    public String getNom() { return nom; }

    // Setter avec validation
    public void setAge(int age) {
        if (age >= 0 && age <= 150) this.age = age;
    }

    // static → accès via Classe.methode()
    public static int getCompteur() { return compteur; }

    // toString()
    @Override
    public String toString() {
        return nom + " (" + age + " ans)";
    }
}

Collections

import java.util.ArrayList;
import java.util.HashMap;
import java.util.Map;

// ArrayList
ArrayList<String> noms = new ArrayList<>();
noms.add("Alice");      // ajouter
noms.get(0);             // lire
noms.set(0, "Bob");     // modifier
noms.remove(0);          // supprimer
noms.size();              // taille
noms.contains("Alice"); // chercher
noms.isEmpty();           // vide ?

// HashMap
HashMap<String, Integer> notes = new HashMap<>();
notes.put("Alice", 18);           // ajouter
notes.get("Alice");                // lire → 18
notes.getOrDefault("Bob", 0);     // lire ou défaut
notes.remove("Alice");             // supprimer
notes.containsKey("Alice");       // clé existe ?

// Parcourir HashMap
for (Map.Entry<String, Integer> e : notes.entrySet()) {
    System.out.println(e.getKey() + " → " + e.getValue());
}
