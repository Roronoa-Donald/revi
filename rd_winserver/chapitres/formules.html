Cheat Sheet Windows Server

Toutes les commandes essentielles et concepts clés en un seul endroit.

PowerShell — Commandes de Base

# Informations système
Get-ComputerInfo                  # Infos complètes du serveur
Rename-Computer "SRV-DC01"       # Renommer le serveur
Restart-Computer                  # Redémarrer
Get-WindowsFeature                # Lister les rôles/fonctionnalités
Install-WindowsFeature AD-Domain-Services -IncludeManagementTools

# Réseau
Get-NetIPAddress                  # Voir les IP
New-NetIPAddress -InterfaceAlias "Ethernet" -IPAddress 192.168.1.10 -PrefixLength 24 -DefaultGateway 192.168.1.1
Set-DnsClientServerAddress -InterfaceAlias "Ethernet" -ServerAddresses 192.168.1.10,192.168.1.11
Test-NetConnection "SRV-DC01" -Port 389

# Services
Get-Service | Where-Object Status -eq "Stopped"
Start-Service "DNS"
Restart-Service "NTDS"

Active Directory

# Installer AD DS et promouvoir en DC
Install-WindowsFeature AD-Domain-Services -IncludeManagementTools
Install-ADDSForest -DomainName "lab.local" -DomainNetbiosName "LAB" -InstallDns

# Créer OU / Utilisateur / Groupe
New-ADOrganizationalUnit -Name "IT" -Path "DC=lab,DC=local"
New-ADUser -Name "Jean Dupont" -SamAccountName "jdupont" -Enabled $true -AccountPassword (ConvertTo-SecureString "P@ss1234" -AsPlainText -Force)
New-ADGroup -Name "GRP_IT" -GroupScope Global
Add-ADGroupMember -Identity "GRP_IT" -Members "jdupont"

# Vérification
Get-ADDomain | Get-ADForest | Get-ADDomainController
netdom query fsmo
repadmin /replsummary | dcdiag /v

GPO

New-GPO -Name "Sécurité Mots de Passe"
New-GPLink -Name "Sécurité Mots de Passe" -Target "DC=lab,DC=local"
gpresult /r             # Résumé des GPO appliquées
gpresult /h rapport.html # Rapport HTML détaillé
gpupdate /force          # Forcer la mise à jour GPO
rsop.msc                 # Resultant Set of Policy

DNS & DHCP

# DNS
Install-WindowsFeature DNS -IncludeManagementTools
Add-DnsServerPrimaryZone -Name "lab.local" -ReplicationScope "Forest"
Add-DnsServerResourceRecordA -ZoneName "lab.local" -Name "srv-web" -IPv4Address 192.168.1.20
nslookup srv-dc01.lab.local | Resolve-DnsName "srv-dc01.lab.local"

# DHCP
Install-WindowsFeature DHCP -IncludeManagementTools
Add-DhcpServerInDC -DnsName "srv-dc01.lab.local"
Add-DhcpServerv4Scope -Name "LAN" -StartRange 192.168.1.51 -EndRange 192.168.1.200 -SubnetMask 255.255.255.0
Set-DhcpServerv4OptionValue -Router 192.168.1.1 -DnsServer 192.168.1.10

Sauvegarde

Install-WindowsFeature Windows-Server-Backup
$policy = New-WBPolicy
Add-WBSystemState -Policy $policy
Set-WBSchedule -Policy $policy -Schedule 22:00
Start-WBBackup -Policy (Get-WBPolicy)
Get-WBBackupSet                # Voir les sauvegardes disponibles
Get-WBSummary                  # Résumé de la dernière

Haute Disponibilité

Install-WindowsFeature Failover-Clustering -IncludeManagementTools
Test-Cluster -Node "SRV-NODE1","SRV-NODE2"
New-Cluster -Name "CLUSTER-PROD" -Node "SRV-NODE1","SRV-NODE2" -StaticAddress 192.168.1.200
Get-ClusterNode | Get-Cluster

# NLB
Install-WindowsFeature NLB -IncludeManagementTools
New-NlbCluster -InterfaceName "Ethernet" -ClusterPrimaryIP 192.168.1.250

Maintenance & Monitoring

# Event Viewer
Get-EventLog -LogName System -EntryType Error -Newest 50
Get-WinEvent -FilterHashtable @{LogName='Security';Id=4625} -MaxEvents 20

# Performance
Get-Counter '\Processor(_Total)\% Processor Time'
Get-Counter '\Memory\Available MBytes'

# WSUS
Install-WindowsFeature UpdateServices -IncludeManagementTools

# Diagnostic
dcdiag /v | repadmin /replsummary | Get-WBSummary

Hyper-V & Snapshots

Install-WindowsFeature Hyper-V -IncludeManagementTools -Restart

# Créer VM
New-VM -Name "SRV-WEB01" -Generation 2 -MemoryStartupBytes 4GB -NewVHDPath "D:\VMs\disk.vhdx" -NewVHDSizeBytes 60GB
Set-VMProcessor -VMName "SRV-WEB01" -Count 2
Start-VM "SRV-WEB01"

# Switches
New-VMSwitch -Name "Production" -NetAdapterName "Ethernet" -AllowManagementOS $true

# Snapshots
Checkpoint-VM -Name "SRV-WEB01" -SnapshotName "Avant-MAJ"
Restore-VMSnapshot -VMName "SRV-WEB01" -Name "Avant-MAJ"
Remove-VMSnapshot -VMName "SRV-WEB01" -Name "Avant-MAJ"

# Export / Import
Export-VM -Name "SRV-WEB01" -Path "E:\Exports"
