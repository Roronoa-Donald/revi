⚡ Bonne nouvelle : Les exceptions sont le concept le PLUS identique entre PHP et C#. Si tu connais try/catch en PHP, tu le connais déjà en C#. Les différences sont minimes.

try / catch / finally

PHP
try {
    $resultat = 10 / 0;
    echo $resultat;
} catch (DivisionByZeroError $e) {
    echo "Erreur : " . $e->getMessage();
} finally {
    echo "\nToujours exécuté !";
}

// Avec une fonction qui throw
function diviser($a, $b) {
    if ($b == 0) {
        throw new Exception("Division par zéro !");
    }
    return $a / $b;
}

try {
    echo diviser(10, 0);
} catch (Exception $e) {
    echo "Erreur : " . $e->getMessage();
    // getMessage() = méthode
}
C#
try {
    int resultat = 10 / 0;
    Console.WriteLine(resultat);
} catch (DivideByZeroException e) {
    Console.WriteLine($"Erreur : {e.Message}");
} finally {
    Console.WriteLine("Toujours exécuté !");
}

// Avec une méthode qui throw
static double Diviser(double a, double b) {
    if (b == 0)
        throw new Exception("Division par zéro !");
    return a / b;
}

try {
    Console.WriteLine(Diviser(10, 0));
} catch (Exception e) {
    Console.WriteLine($"Erreur : {e.Message}");
    // Message = propriété (pas de parenthèses)
}
💡 Seule différence :
• PHP: $e->getMessage() — c'est une méthode (avec parenthèses)
• C#: e.Message — c'est une propriété (sans parenthèses)

Exceptions personnalisées

Créer ses propres exceptions pour des erreurs métier spécifiques. C'est de l'héritage appliqué !

PHP
// Exception personnalisée
class PrixInvalideException extends Exception {
    private $prixTente;

    public function __construct($prix) {
        parent::__construct(
            "Prix invalide : $prix"
        );
        $this->prixTente = $prix;
    }

    public function getPrixTente() {
        return $this->prixTente;
    }
}

class StockInsuffisantException extends Exception {
    public function __construct($dispo, $demande) {
        parent::__construct(
            "Stock insuffisant : $dispo dispo, "
            . "$demande demandé"
        );
    }
}

// Utilisation
class Produit {
    private $prix;
    private $stock;

    public function __construct($prix, $stock) {
        $this->setPrix($prix);
        $this->stock = $stock;
    }

    public function setPrix($p) {
        if ($p < 0) {
            throw new PrixInvalideException($p);
        }
        $this->prix = $p;
    }

    public function vendre($qte) {
        if ($qte > $this->stock) {
            throw new StockInsuffisantException(
                $this->stock, $qte
            );
        }
        $this->stock -= $qte;
    }
}

// Multi-catch
try {
    $p = new Produit(25, 10);
    $p->vendre(15);
} catch (PrixInvalideException $e) {
    echo "Prix: " . $e->getMessage();
} catch (StockInsuffisantException $e) {
    echo "Stock: " . $e->getMessage();
} catch (Exception $e) {
    echo "Autre: " . $e->getMessage();
}
C#
// Exception personnalisée
class PrixInvalideException : Exception {
    public double PrixTente { get; }

    public PrixInvalideException(double prix)
        : base($"Prix invalide : {prix}")
    {
        PrixTente = prix;
    }
}

class StockInsuffisantException : Exception {
    public StockInsuffisantException(int dispo, int demande)
        : base($"Stock insuffisant : {dispo} dispo, "
             + $"{demande} demandé")
    { }
}

// Utilisation
class Produit {
    private double prix;
    private int stock;

    public Produit(double prix, int stock) {
        Prix = prix; // passe par set
        this.stock = stock;
    }

    public double Prix {
        get => prix;
        set {
            if (value < 0)
                throw new PrixInvalideException(value);
            prix = value;
        }
    }

    public void Vendre(int qte) {
        if (qte > stock)
            throw new StockInsuffisantException(stock, qte);
        stock -= qte;
    }
}

// Multi-catch (catch séparés)
try {
    Produit p = new Produit(25, 10);
    p.Vendre(15);
} catch (PrixInvalideException e) {
    Console.WriteLine($"Prix: {e.Message}");
} catch (StockInsuffisantException e) {
    Console.WriteLine($"Stock: {e.Message}");
} catch (Exception e) {
    Console.WriteLine($"Autre: {e.Message}");
}
⚡ Différences mineures :
• PHP: extends Exception / C#: : Exception (comme pour l'héritage normal)
• PHP: parent::__construct($msg) / C#: : base($msg)
• PHP: catch (A | B $e) multi-catch en 1 bloc / C#: catch séparés
• La logique est 100% identique

Pattern complet — Classe + Encapsulation + Exceptions

Le schéma qui combine TOUT ce que tu as appris (examen exercice 4) :

PHP — Pattern complet
// 1. Exception custom
class NoteInvalideException extends Exception {
    public function __construct($note) {
        parent::__construct(
            "Note invalide : $note (0-20 requis)"
        );
    }
}

// 2. Classe encapsulée
class Etudiant {
    private $nom;
    private $notes = [];

    public function __construct($nom) {
        $this->nom = $nom;
    }

    // 3. Setter avec exception
    public function ajouterNote($note) {
        if ($note < 0 || $note > 20) {
            throw new NoteInvalideException($note);
        }
        $this->notes[] = $note;
    }

    public function moyenne() {
        if (empty($this->notes)) return 0;
        return array_sum($this->notes)
             / count($this->notes);
    }
}

// 4. Try/catch
try {
    $e = new Etudiant("Alice");
    $e->ajouterNote(15);
    $e->ajouterNote(25); // ❌ Exception
} catch (NoteInvalideException $ex) {
    echo $ex->getMessage();
}
C# — Pattern complet
// 1. Exception custom
class NoteInvalideException : Exception {
    public NoteInvalideException(double note)
        : base($"Note invalide : {note} (0-20 requis)")
    { }
}

// 2. Classe encapsulée
class Etudiant {
    public string Nom { get; private set; }
    private List<double> notes = new();

    public Etudiant(string nom) {
        Nom = nom;
    }

    // 3. Méthode avec exception
    public void AjouterNote(double note) {
        if (note < 0 || note > 20)
            throw new NoteInvalideException(note);
        notes.Add(note);
    }

    public double Moyenne() {
        if (notes.Count == 0) return 0;
        return notes.Average();
    }
}

// 4. Try/catch
try {
    Etudiant e = new Etudiant("Alice");
    e.AjouterNote(15);
    e.AjouterNote(25); // ❌ Exception
} catch (NoteInvalideException ex) {
    Console.WriteLine(ex.Message);
}

Exercice examen — Gestion d'erreurs (PHP + C#)

Consigne (inspirée examen 2024/2025, exercice 4 — 5 pts) :
1. Crée une exception SoldeInsuffisantException avec montant tenté et solde actuel
2. Crée une classe CompteBancaire avec solde (private), getter, méthode retirer()
3. retirer() lance l'exception si montant > solde
4. Dans le programme principal, crée un compte avec 500€ et tente de retirer 800€
5. Attrape l'exception et affiche un message clair
PHP
<?php
class SoldeInsuffisantException extends Exception {
    private $montant;
    private $solde;

    public function __construct($montant, $solde) {
        parent::__construct(
            "Retrait de {$montant}€ impossible "
            . "(solde: {$solde}€)"
        );
        $this->montant = $montant;
        $this->solde = $solde;
    }
}

class CompteBancaire {
    private $solde;

    public function __construct($solde) {
        $this->solde = $solde;
    }

    public function getSolde() {
        return $this->solde;
    }

    public function retirer($montant) {
        if ($montant > $this->solde) {
            throw new SoldeInsuffisantException(
                $montant, $this->solde
            );
        }
        $this->solde -= $montant;
    }
}

// Test
try {
    $compte = new CompteBancaire(500);
    echo "Solde : {$compte->getSolde()}€\n";
    $compte->retirer(200); // ✅ OK
    echo "Après retrait : {$compte->getSolde()}€\n";
    $compte->retirer(800); // ❌ Exception
} catch (SoldeInsuffisantException $e) {
    echo "❌ " . $e->getMessage();
}
C#
class SoldeInsuffisantException : Exception {
    public double Montant { get; }
    public double SoldeActuel { get; }

    public SoldeInsuffisantException(
        double montant, double solde)
        : base($"Retrait de {montant}€ impossible "
             + $"(solde: {solde}€)")
    {
        Montant = montant;
        SoldeActuel = solde;
    }
}

class CompteBancaire {
    public double Solde { get; private set; }

    public CompteBancaire(double solde) {
        Solde = solde;
    }

    public void Retirer(double montant) {
        if (montant > Solde)
            throw new SoldeInsuffisantException(
                montant, Solde);
        Solde -= montant;
    }
}

// Test
try {
    var compte = new CompteBancaire(500);
    Console.WriteLine($"Solde : {compte.Solde}€");
    compte.Retirer(200); // ✅
    Console.WriteLine($"Après : {compte.Solde}€");
    compte.Retirer(800); // ❌
} catch (SoldeInsuffisantException e) {
    Console.WriteLine($"❌ {e.Message}");
}
Points examen : Exception custom héritant d'Exception (2 pts) + Constructeur avec parent::/base() (1 pt) + throw dans la bonne méthode (1 pt) + try/catch dans le main (1 pt) = 5 pts. Identique en PHP et C# !
