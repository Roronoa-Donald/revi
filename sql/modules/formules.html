DDL Définition de Données

Créer une Base
CREATE DATABASE MaBase;
Créer une Table
CREATE TABLE Clients (
id INT PRIMARY KEY IDENTITY,
nom VARCHAR(100) NOT NULL,
email VARCHAR(200) UNIQUE
);
Modifier une Table
ALTER TABLE Clients
ADD telephone VARCHAR(20);

ALTER TABLE Clients
DROP COLUMN telephone;
Supprimer une Table
DROP TABLE Clients;
-- Suppression définitive !
Vider une Table
TRUNCATE TABLE Clients;
-- Vide sans log, rapide

CONTRAINTES Intégrité des Données

Clé Primaire
id INT PRIMARY KEY IDENTITY(1,1)
Clé Étrangère
FOREIGN KEY (client_id)
REFERENCES Clients(id)
ON DELETE CASCADE
UNIQUE
email VARCHAR(200) UNIQUE
CHECK
age INT CHECK (age >= 0 AND age <= 150)
DEFAULT
statut VARCHAR(20) DEFAULT 'actif'
NOT NULL
nom VARCHAR(100) NOT NULL

DML Manipulation de Données

INSERT
INSERT INTO Clients (nom, email)
VALUES ('Alice', 'alice@test.com');
UPDATE
UPDATE Clients
SET nom = 'Bob'
WHERE id = 1;
DELETE
DELETE FROM Clients
WHERE id = 1;
-- ⚠️ Toujours mettre WHERE !
INSERT multi-lignes
INSERT INTO Clients (nom)
VALUES ('A'), ('B'), ('C');
MERGE (Upsert)
MERGE INTO Cible AS t
USING Source AS s
ON t.id = s.id
WHEN MATCHED THEN UPDATE...
WHEN NOT MATCHED THEN INSERT...;

SELECT Interrogation

Tout sélectionner
SELECT * FROM Clients;
WHERE (Filtres)
WHERE age > 18
WHERE nom LIKE 'A%'
WHERE id IN (1, 2, 3)
WHERE age BETWEEN 18 AND 65
ORDER BY
ORDER BY nom ASC
ORDER BY age DESC, nom ASC
JOIN
SELECT c.nom, o.total
FROM Clients c
INNER JOIN Orders o
ON c.id = o.client_id;
Types de JOIN
INNER JOIN → correspondances
LEFT JOIN → tout à gauche
RIGHT JOIN → tout à droite
FULL JOIN → tout des deux
CASE WHEN
SELECT nom,
CASE
WHEN age >= 18 THEN 'Majeur'
ELSE 'Mineur'
END AS statut
FROM Clients;
DISTINCT
SELECT DISTINCT ville FROM Clients;
TOP / OFFSET
SELECT TOP 10 * FROM Clients;

ORDER BY id
OFFSET 20 ROWS
FETCH NEXT 10 ROWS ONLY;

AGRÉGATS Calculs & Groupement

Fonctions d'agrégat
COUNT(*), COUNT(col)
SUM(col), AVG(col)
MIN(col), MAX(col)
GROUP BY
SELECT ville, COUNT(*) AS nb
FROM Clients
GROUP BY ville;
HAVING (filtre groupes)
SELECT ville, COUNT(*) AS nb
FROM Clients
GROUP BY ville
HAVING COUNT(*) > 5;

AVANCÉ Procédures, Fonctions & Triggers

Procédure Stockée
CREATE PROCEDURE GetUser
@id INT
AS
BEGIN
SELECT * FROM Users
WHERE id = @id
END

EXEC GetUser @id = 1;
Paramètre OUTPUT
CREATE PROCEDURE CountUsers
@total INT OUTPUT
AS
BEGIN
SELECT @total = COUNT(*)
FROM Users
END
Trigger
CREATE TRIGGER trg_Log
ON Users AFTER INSERT
AS
BEGIN
INSERT INTO Logs(action)
SELECT 'New: ' + name
FROM inserted
END
INSTEAD OF
CREATE TRIGGER trg_NoDelete
ON Users INSTEAD OF DELETE
AS
BEGIN
UPDATE Users
SET actif = 0
WHERE id IN
(SELECT id FROM deleted)
END
Tables Magiques
inserted → lignes ajoutées
deleted → lignes supprimées

Dispo dans les Triggers
Vue (VIEW)
CREATE VIEW v_ActiveUsers AS
SELECT * FROM Users
WHERE actif = 1;

SELECT * FROM v_ActiveUsers;

BONUS Astuces & Patterns

Sous-requête
SELECT * FROM Clients
WHERE id IN (
SELECT client_id
FROM Orders
WHERE total > 100
);
CTE (WITH)
WITH TopClients AS (
SELECT client_id,
SUM(total) AS sum_total
FROM Orders
GROUP BY client_id
)
SELECT * FROM TopClients
WHERE sum_total > 1000;
Window Functions
SELECT nom, salaire,
RANK() OVER (
ORDER BY salaire DESC
) AS rang,
SUM(salaire) OVER () AS total
FROM Employes;
Index
CREATE INDEX idx_nom
ON Clients(nom);

-- Index composé
CREATE INDEX idx_ville_nom
ON Clients(ville, nom);
EXISTS
SELECT * FROM Clients c
WHERE EXISTS (
SELECT 1 FROM Orders o
WHERE o.client_id = c.id
);

2. WHERE
3. GROUP BY
4. HAVING
5. SELECT
6. ORDER BY
7. TOP / OFFSET
